Integration Example

// Dashboard Component Example
import React from 'react';
import LineChart from './assets/dashboard/charts/line-chart.svg';
import CircularGauge from './assets/dashboard/widgets/circular-gauge.svg';
import SystemStatus from './assets/dashboard/panels/system-status.svg';

function Dashboard() {
  return (
    <div className="dashboard-container">
      <div className="metrics-row">
        <img src={CircularGauge} alt="CPU Usage" />
        <img src={SystemStatus} alt="System Health" />
      </div>
      <div className="charts-section">
        <img src={LineChart} alt="Performance Metrics" />
      </div>
    </div>
  );
}

// Dynamic SVG Loading
const loadDashboardElement = async (elementType) => {
  const elementMap = {
    'gauge': './dashboard/widgets/circular-gauge.svg',
    'stats': './dashboard/widgets/stat-card.svg',
    'stream': './dashboard/indicators/data-stream.svg'
  };
  
  return await fetch(elementMap[elementType]);
};
