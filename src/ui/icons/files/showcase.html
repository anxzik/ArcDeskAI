// Example: Using SVG assets in Electron app

// In your renderer process
const aiModelIcon = document.createElement('img');
aiModelIcon.src = './assets/icons/openai-gpt.svg';
aiModelIcon.className = 'ai-model-icon';

// For React components
import GPTIcon from './assets/icons/openai-gpt.svg';
